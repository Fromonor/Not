task.spawn(function()
-- Variables
local player = game.Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()
local torso = character:WaitForChild("Torso")

-- Clone the Ring MeshPart from ReplicatedStorage
local ring = game.ReplicatedStorage.Resources.KJEffects.Ring:Clone()

-- Set its properties
ring.Anchored = true
ring.CanCollide = false
ring.Size = Vector3.new(6.5, 0.5, 6.5)

-- Parent the cloned ring to the player's character
ring.Parent = game.Players.LocalPlayer.Character

-- Position the ring (adjust as needed)
ring.Position = game.Players.LocalPlayer.Character["Torso"].Position + Vector3.new(0, 0.7, 0)

-- Function to grow and fade out the part over 0.3 seconds
local function growAndFadeOut(part)
    local startSize = ring.Scale
    local endSize = Vector3.new(9, 9, 0.1) -- Final size of the ring (larger than initial)
    local duration = 0.14 -- Duration in seconds
    local step = 0.02 -- Step size for the loop
    local steps = duration / step

    for i = 0, steps do
        local alpha = i / steps
        -- Calculate size and transparency
        ring.Scale = startSize:Lerp(endSize, alpha)
        part.Transparency = alpha
        wait(step)
    end
    part:Destroy()
end
end)

task.spawn(function()
    local player = game.Players.LocalPlayer
    local character = player.Character or player.CharacterAdded:Wait()
    local humanoidRootPart = character:WaitForChild("HumanoidRootPart")

    local teleportDistance = 2.2 -- The distance to teleport forward each step
    local teleportInterval = 0 -- Time interval between each teleport (in seconds)
    local totalDuration = 0.62 -- Total duration of the teleportation (in seconds)

    local timeElapsed = 0
    local startTime = tick() -- Record the start time

    while timeElapsed < totalDuration do
        humanoidRootPart.CFrame = humanoidRootPart.CFrame * CFrame.new(0, 0, -teleportDistance)
        wait(teleportInterval)
        timeElapsed = tick() - startTime -- Update the elapsed time
    end
end)
